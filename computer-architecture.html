ALU (32-bit) — Structural Verilog

Built an ALU supporting AND, ADD, SUB, SLT, SRL, SRA, SLL with an 8:1 result mux and standard flags. (This matches the “final 8:1 result mux” mapping: AND/ADD/SUB/SLT/SRL/SRA/SLL.) :contentReference[oaicite:0]{index=0}

Verilog / SystemVerilog Carry-Lookahead Adder (CLA) Shifters: SRL/SRA/SLL Flags: zero/equal/overflow 
`include "alu.svh"

module STUDENT_alu(
  input  logic [31:0] x,
  input  logic [31:0] y,
  input  logic [2:0]  op,
  output logic [31:0] z,
  output logic        zero, equal, overflow
);

  logic [31:0] z_and, z_add, z_sub, z_slt, z_srl, z_sra, z_sll;
  logic cout_add, c31_add;
  logic cout_sub, c31_sub, ov_sub;

  // AND
  and32 and_unit (.x(x), .y(y), .z(z_and));

  // ADD via CLA (needs cout + c31 exposed by CLA) :contentReference[oaicite:1]{index=1}
  CLA add_unit_cla (.a(x), .b(y), .cin(1'b0), .s(z_add), .cout(cout_add), .c31(c31_add));

  // SUB via CLA: x + ~y + 1  (build ~y structurally) :contentReference[oaicite:2]{index=2}
  logic [31:0] y_inv;
  genvar i;
  generate
    for (i=0; i<32; i++) begin : invert_y
      not n (y_inv[i], y[i]);
    end
  endgenerate
  CLA sub_unit_cla (.a(x), .b(y_inv), .cin(1'b1), .s(z_sub), .cout(cout_sub), .c31(c31_sub));

  // Overflow for SUB feeds SLT :contentReference[oaicite:3]{index=3}
  overflow ov_sub_u (.c31(c31_sub), .cout(cout_sub), .ov(ov_sub));

  // SLT uses SUB result + SUB overflow :contentReference[oaicite:4]{index=4}
  slt32 slt_u (.z_sub(z_sub), .ov_sub(ov_sub), .z_slt(z_slt));

  // Shifters (already implemented structurally in your project) :contentReference[oaicite:5]{index=5}
  srl srl_u (.x(x), .y(y), .z(z_srl));
  sra sra_u (.x(x), .y(y), .z(z_sra));
  sll sll_u (.x(x), .y(y), .z(z_sll));

  // Select ALU overflow based on op using 1-bit mux trees :contentReference[oaicite:6]{index=6}
  logic c31_sel, cout_sel;
  mux8_1bit_tree c31_mux (
    .D0(1'b0), .D1(c31_add), .D2(c31_sub), .D3(1'b0),
    .D4(1'b0), .D5(1'b0),    .D6(1'b0),    .D7(1'b0),
    .S(op), .Y(c31_sel)
  );
  mux8_1bit_tree cout_mux (
    .D0(1'b0), .D1(cout_add), .D2(cout_sub), .D3(1'b0),
    .D4(1'b0), .D5(1'b0),     .D6(1'b0),     .D7(1'b0),
    .S(op), .Y(cout_sel)
  );
  overflow ov_one (.c31(c31_sel), .cout(cout_sel), .ov(overflow));

  // zero/equal flags :contentReference[oaicite:7]{index=7}
  zero32 zero_on_result (.z(z), .zero(zero));
  equal  eq_xy_u        (.a(x), .b(y), .equal(equal));

  // Final 8:1 result mux mapping :contentReference[oaicite:8]{index=8}
  mux8_32bit_tree result_mux (
    .D0(z_and), .D1(z_add), .D2(z_sub), .D3(z_slt),
    .D4(z_srl), .D5(z_sra), .D6(z_sll), .D7(32'b0),
    .S(op), .Y(z)
  );

endmodule
